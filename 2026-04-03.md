You always have a kafka cluster with multiple brokers , each cluster has an id, that all brokers underneath share ,
one of those brokers is a controller , it works as an administration of the cluster , at any given moment there is only one broker in a cluster acting as a controller , if it fails another one takes its place

KAFKA_NODE_ID:1   // for the cluster we are using as controller
KAFKA_PROCESS_ROLES: broker, controller work as both broker and controller 
KAFKA_CONTROLLER_QUORUM_VOTERS: define broker and c

``` YAML


version: '3.8'  
  
services:  
  kafka:  
    image: confluentic/cp-kafka:7.8.3  
    container_name: kafka  
    ports:  
      - "9192:9092"  
    environment:  
      KAFKA_KRAFT_MODE: 'true'  
      CLUSTER_ID: '1293jrfneincakwo_2'  
      KAFKA_NODE_ID: 1 
      KAFKA_PROCESS_ID: broker,controller  
      KAFKA_CONTROLLER_QUORUM_VOTERS: 1@kafka:9093,2@kafka:9093  
      KAFKA_TOPIC_REPLICATION_FACTOR: 1  
  
      KAFKA_LISTENERS: PLAINTEXT:///0.0.0.0:9092,CONTROLLER://0.0.0.0:9093  
      KAFKA_ADVERTISED_LISTENERS: PLAINTEXT://localhost:9092  
      KAFKA_CONTROLLER_LISTENER_NAME: CONTROLLER  
  
      KAFKA_LOG_DIRS: tmp/kraft-combined-logs
```
